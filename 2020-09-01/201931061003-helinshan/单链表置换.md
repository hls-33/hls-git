##1.单链表置换

``` javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var reverseList = function(head) {
     var list = head;
     var p = list;
     var q = null;
    if(!p)
    {
        return null;
    }
    while(p.next !== null)
    {
        q = p.next;
        p.next = q.next;
        q.next = list;
        list = q;     
    }
    return list;
}
```
